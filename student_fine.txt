begin
declare vfinshined int default 0;

declare s float;
declare st_cursor cursor for select * from info;
declare continue handler for not found set vfinshined=1;
open st_cursor;
setdata:loop
     fetch st_cursor into i,n,s;
     if vfinshined=1 then
	 leave setdata;
	 end if;

     end loop setdata;
